const ll MAXN = 1e7;
const ll MOD = 1e9+7;

ll n,x;
vl a(MAXN),dp(MAXN);
void solution(){
  cin>>n>>x;
  a.assign(n+1,0);
  dp.assign(x+1,0);
  for (ll i=1;i<=n;i++) cin>>a[i];
  dp[0]=1;
  for (ll i=1;i<=x;i++){
    for (ll j=1;j<=n;j++){
      if (i>=a[j]) dp[i]=(dp[i]+dp[i-a[j]]%MOD)%MOD;
    }
  }
  cout<<dp[x]<<endl;
}
